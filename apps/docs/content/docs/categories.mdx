The `categories` module provides methods to interact with the categories API.

## getCategories

Get all contents from the workspace.

<Tabs items={['Basic', 'With parameters', 'With Pagination']}>
  <Tab value="Basic">
```typescript
import { postivaClient } from "@/libs/postiva";

async function getCategories() {
  try {
    const categories = await postivaClient.categories.getCategories();
    console.log(categories);
  } catch (error) {
    console.error('Error fetching categories:', error);
  }
}
```
  </Tab>
  <Tab value="With parameters">
```typescript
import { postivaClient } from "@/libs/postiva";
import { GetCategoriesType } from '@postiva/client'

async function getCategoriesWithQuery({ query }: GetCategoriesType) {
  try {
    const categories = await postivaClient.categories.getCategories({ query });
    console.log(categories);
  } catch (error) {
    console.error('Error fetching categories:', error);
  }
}
```

## getCategoryById

Get a category by its ID.

```typescript
import { postivaClient } from "@/libs/postiva";

async function getCategoryById(id: string) {
  try {
    const category = await postivaClient.categories.getCategoryById(id);
    console.log(category);
  } catch (error) {
    console.error('Error fetching category:', error);
  }
}
```
